# Get pods status
kubectl get pods -n rag-system

# Get logs:
kubectl logs -n rag-system deployment/generation-service --tail=20

# UI port forwarding for access UI in dev
kubectl port-forward -n rag-system service/frontend-service 8501:8501

# Delete namespaces
kubectl delete namespace rag-system

# Restart a specific container
kubectl rollout restart deployment/query-service -n rag-system

# Rebuild a specific docker
docker build -t frontend-service:dev services/frontend-service/   

# Exceute a specific command
kubectl exec -n rag-system deployment/generation-service -- cat logs/llm_usage.log

# check PVCs
kubectl get pvc -n rag-system

# Complete Minikube reset (fastest way to break the cycle)
minikube stop
minikube delete
minikube start --memory=8192 --cpus=4
